## latest posts

<Divider />

<SourceList filter="posts">
  {(posts) => (
    <Flex
      sx={{
        flexWrap: "wrap",
        ml: (theme) => `-${theme.space[2]}px`,
        mr: (theme) => `-${theme.space[2]}px`,
      }}
    >
      {posts
        .filter((edge) => !edge.node.frontmatter.isPrivate)
        .reduce((posts, post) => {
          return post.node.frontmatter.pinned
            ? [post, ...posts]
            : [...posts, post]
        }, [])
        .splice(0, 4)
        .map((edge, index) => {
          const {
            frontmatter: {
              title,
              featuredImageUrl,
              tags,
              date,
              dateModified,
              pinned,
            },
            excerpt,
            fields: { slug },
          } = edge.node
          return (
            <PostCard
              key={index}
              title={title}
              featuredImageUrl={featuredImageUrl}
              tags={tags}
              date={date}
              dateModified={dateModified}
              excerpt={excerpt}
              slug={slug}
              pinned={pinned}
            />
          )
        })}
    </Flex>
  )}
</SourceList>
