<Divider />

<SourceList filter="posts">
  {(posts) => (
    <Flex
      sx={{
        flexWrap: "wrap",
        ml: (theme) => `-${theme.space[2]}px`,
        mr: (theme) => `-${theme.space[2]}px`,
      }}
    >
      {posts
        .filter((edge) => !edge.node.frontmatter.isPrivate)
        .sort(
          (a, b) =>
            new Date(b.node.frontmatter.date) -
            new Date(a.node.frontmatter.date)
        )
        .splice(0, 4)
        .map((edge, index) => {
          const {
            frontmatter: { title, featuredImage, tags, date },
            excerpt,
            fields: { slug },
          } = edge.node
          return (
            <PostCard
              key={index}
              title={title}
              featuredImage={featuredImage}
              tags={tags}
              date={date}
              excerpt={excerpt}
              slug={slug}
            />
          )
        })}
    </Flex>
  )}
</SourceList>
