## tags by usage

<Grid sx={{gridTemplateColumns: ['1fr', '1fr 1fr'], gridGap: 3}}>

<SourceTags>
  {(source) => (
    <Grid>
      {source
        .sort((a, b) => b.count - a.count)
        .splice(0, 5)
        .map((tag, index) => {
          const { name, count } = tag
          return (
            <Flex key={index} sx={{ alignItems: "center" }}>
              <Text sx={{ color: "primary", fontWeight: "bold", mr: 1 }}>{`${
                index + 1
              }:`}</Text>
              <Text sx={{ mr: 2 }}>{name}</Text>
              <Flex
                sx={{
                  alignItems: "center",
                  backgroundColor: "surface",
                  borderRadius: "50%",
                  color: "success",
                  fontWeight: "bold",
                  fontSize: 0,
                  justifyContent: "center",
                  width: "28px",
                  height: "28px",
                }}
              >
                {`x${count}`}
              </Flex>
            </Flex>
          )
        })}
    </Grid>
  )}
</SourceTags>

</Grid>
