import { WritingCard } from "../components/WritingCard"

# writing

Here's a collection of articles I've written that have been officially published ☺️

<SourceList filter="writing">
  {(writing) => (
    <Flex
      sx={{
        flexWrap: "wrap",
        ml: (theme) => `-${theme.space[2]}px`,
        mr: (theme) => `-${theme.space[2]}px`,
      }}
    >
      {writing
        .filter((edge) => !edge.node.frontmatter.isPrivate)
        .map((edge, index) => {
          const {
            frontmatter: { title, featuredImage, tags, date, url, misc, author },
            excerpt,
      
          } = edge.node
          return (
            <WritingCard
              key={index}
              title={title}
              featuredImage={featuredImage}
              tags={tags}
              date={date}
              excerpt={excerpt}
              url={url}
              misc={misc}
              author={author}
            />
          )
        })}
    </Flex>
)}

</SourceList>
