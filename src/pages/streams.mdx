import { StreamCard } from '../components/stream-card'

# streams

I prefer to write, but you can occasionally find me on stream.

<Divider />

<SourceList filter="stream">
  {(stream) => (
    <Grid
      sx={{
        gap: 4
      }}
    >
      {stream
        .filter((edge) => !edge.node.frontmatter.isPrivate)
        .map((edge, index) => {
          const {
            frontmatter: { title, featuredImage, tags, date, url, misc, author },
            excerpt,
      
          } = edge.node
          return (
            <StreamCard
              key={index}
              title={title}
              featuredImage={featuredImage}
              tags={tags}
              date={date}
              excerpt={excerpt}
              url={url}
              misc={misc}
              author={author}
            />
          )
        })}
    </Grid>
)}

</SourceList>
